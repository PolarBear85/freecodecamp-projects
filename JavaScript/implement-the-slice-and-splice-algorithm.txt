function frankenSplice(array1,array2,index) {
    
    let newSplice = [...array2];
    let spliceIndex=array1.length;
    
    for (let i = 0 ;i <spliceIndex; i++){
        newSplice.splice(index,0,array1[i]);
        index++;
    }

}
